import express from 'express' import fs from 'fs-extra' import pino from 'pino' import pn from 'awesome-phonenumber' import { makeWASocket, useMultiFileAuthState, delay, makeCacheableSignalKeyStore, Browsers, jidNormalizedUser, fetchLatestBaileysVersion } from '@whiskeysockets/baileys' import { upload } from './mega.js' const router = express.Router() async function removeFile(path) { if (fs.existsSync(path)) await fs.remove(path) } router.get('/', async (req, res) => { let num = req.query.number const sessionPath = './session' await removeFile(sessionPath) num = num.replace(/[^0-9]/g, '') const phone = pn('+' + num) if (!phone.isValid()) { return res.json({ code: 'INVALID NUMBER' }) } num = phone.getNumber('e164').replace('+', '') const { state, saveCreds } = await useMultiFileAuthState(sessionPath) const { version } = await fetchLatestBaileysVersion() const sock = makeWASocket({ version, auth: { creds: state.creds, keys: makeCacheableSignalKeyStore(state.keys, pino({ level: 'fatal' })) }, logger: pino({ level: 'fatal' }), browser: Browsers.windows('Chrome') }) sock.ev.on('connection.update', async ({ connection }) => { if (connection === 'open') { const file = `${sessionPath}/creds.json` if (!fs.existsSync(file)) return const link = await upload(fs.createReadStream(file), 'session.json') const user = jidNormalizedUser(num + '@s.whatsapp.net') await sock.sendMessage(user, { text: `🔥 SESSION ID 🔥\n\n${link}` }) await removeFile(sessionPath) } }) if (!sock.authState.creds.registered) { await delay(1500) let code = await sock.requestPairingCode(num) code = code?.match(/.{1,4}/g)?.join('-') res.json({ code }) } sock.ev.on('creds.update', saveCreds) }) export default router
